INSERTION SORT


#include <stdio.h>
using namespace std;
void insertionSort(int A[],int n)
{
for(int i=1;i {
int j=i-1;
int key=A[i];
while(j>=0 && A[j]>key)
{
A[j+1]=A[j];
j--;
}
A[j+1]=key;
}
}
int main()
{
int n;
cout<<"Enter number of elements\n";
cin>>n;
int A[n];
cout<<"Enter elements one one\n";
for(int i=0;i {
cin>>A[i];
}
insertionSort(A,n);
cout<<"Sorted Array is\n";
for(int i=0;i< n ;i++)
{
cout<< A[i]<<" ";
}
}
